class Solution {
    public void moveZeroes(int[] nums) {
        int pos = 0;

        // Move non-zero elements forward
        for (int num : nums) {
            if (num != 0) {
                nums[pos++] = num;
            }
        }

        // Fill remaining with zeros
        while (pos < nums.length) {
            nums[pos++] = 0;
        }
    }
}
