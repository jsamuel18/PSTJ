public static int digitSum(int number, String opt) {
    int sum = 0;

    number = Math.abs(number);

    while (number > 0) {
        int digit = number % 10;

        if (opt.equalsIgnoreCase("even") && digit % 2 == 0)
            sum += digit;

        if (opt.equalsIgnoreCase("odd") && digit % 2 != 0)
            sum += digit;

        number /= 10;
    }

    return sum;
}
