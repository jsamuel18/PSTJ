import java.util.*;
import java.util.stream.*;

class Sam {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        List<Integer> nums = new ArrayList<>();

        for (int i = 0; i < n; i++)
            nums.add(sc.nextInt());

        int result = nums.stream()
                .distinct()
                .sorted(Comparator.reverseOrder())
                .skip(1)
                .findFirst()
                .orElse(-1);

        System.out.println(result);
    }
}
